    function userFilterMatch(user, readiness, filter) {
      const enabled = String(user.enabled ?? '1') === '1';
      const status = String(readiness.status || 'UNKNOWN').toLowerCase();
      if (filter === 'all') return true;
      if (filter === 'disabled') return !enabled;
      if (!enabled) return false;
      return status === filter;
    }

    function routeCell(status, detail='', good=false, warn=false) {
      const cls = good ? 'ok' : warn ? 'warn' : 'bad';
      return `<div class="matrix-cell"><span class="chip ${cls}">${esc(status)}</span><small>${esc(detail || '')}</small></div>`;
    }
    function serviceCell(egressId, serviceId, speedItems, matrixItems) {
      const speed = speedItems[egressId] || {};
      const item = matrixItems[egressId] || {};
      const svc = (item.services || {})[serviceId] || {};
      const mbps = speed.server_v7_mbps ? `${speed.server_v7_mbps} Mbps` : '';
      if (!Object.keys(svc).length) return routeCell('NOT_TESTED', mbps || 'no sample', false, true);
      const detail = [`http ${svc.http_code || 'n/a'}`, svc.first_byte_sec !== undefined ? `${svc.first_byte_sec}s first byte` : '', mbps].filter(Boolean).join(' · ');
      return routeCell(svc.ok ? 'OK' : 'FAIL', detail, !!svc.ok, !svc.ok);
    }
    function classCell(channelId, classId, d, speedItems, matrixItems) {
      const trusted = d.trusted_ru || {};
      const decision = trusted.decision || {};
      const speed = speedItems[channelId] || {};
      const mbps = speed.server_v7_mbps ? `${speed.server_v7_mbps} Mbps` : '';
      if (channelId === 'CLIENT_DIRECT') {
        if (classId === 'DIRECT_RU' || classId === 'TRUSTED_RU_SENSITIVE' || classId === 'BANKS_RU' || classId === 'RU_SOCIAL' || classId === 'MARKETPLACES_RU') {
          return routeCell('CLIENT_DIRECT', 'filtered before V7', true, false);
        }
        return routeCell('BYPASS_V7', 'not used for global/video', false, true);
      }
      if (channelId === 'DIRECT_RU') {
        if (classId === 'DIRECT_RU' || classId === 'RU_SOCIAL' || classId === 'MARKETPLACES_RU') {
          return routeCell(d.summary?.direct_ru_route_ok ? 'OK' : 'CHECK', 'V7 public route', !!d.summary?.direct_ru_route_ok, !d.summary?.direct_ru_route_ok);
        }
        if (classId === 'TRUSTED_RU_SENSITIVE' || classId === 'BANKS_RU') {
          const total = Number(trusted.total || 0);
          const localOk = Number(trusted.local_ok || 0);
          const ok = total > 0 && localOk === total;
          return routeCell(ok ? 'OK' : 'DEPENDS', total ? `${localOk}/${total} direct samples` : 'needs samples', ok, !ok);
        }
        return routeCell('N/A', 'direct route class', false, true);
      }
      if (classId === 'VIDEO_OPTIMIZED') return serviceCell(channelId, 'youtube', speedItems, matrixItems);
      if (classId === 'GLOBAL_STABLE') return serviceCell(channelId, 'telegram', speedItems, matrixItems);
      if (classId === 'GLOBAL_FAST' || classId === 'LOW_LATENCY') {
        const item = matrixItems[channelId] || {};
        const ok = item.status === 'OK';
        const status = item.total ? `${item.ok_count || 0}/${item.total}` : 'NOT_TESTED';
        return routeCell(ok ? 'OK' : status, mbps || 'service matrix/speed sample', ok, !ok);
      }
      if (classId === 'TRUSTED_RU_SENSITIVE' || classId === 'BANKS_RU') {
        const candidate = decision.current_candidate || '';
        const uses = candidate === channelId || (channelId === 'vless' && Number(trusted.temporary_ok || 0) > 0);
        return routeCell(uses ? 'PARTIAL' : 'UNKNOWN', `${decision.candidate_result || 'needs route proof'} ${mbps}`.trim(), false, true);
      }
      if (classId === 'DIRECT_RU' || classId === 'RU_SOCIAL' || classId === 'MARKETPLACES_RU') {
        return routeCell('RISK', `external VPN path ${mbps}`.trim(), false, true);
      }
      return routeCell('UNKNOWN', mbps || 'no data', false, true);
    }

    function renderConnectSupport(identity) {
      const attempts = identity.onboarding_attempts || [];

      const reasonFilter = (document.getElementById('identityAttemptFilterReason')?.value || '').trim();
      const allFailures = attempts.filter(a => a.result === 'FAIL');
      const reasonOptions = Array.from(new Set(allFailures.map(a => a.reason || 'unknown'))).sort();
      const filters = `<div class="filters"><label><span class="label">Phone</span><input id="identityAttemptFilterPhone" value="${esc(phoneFilter)}" placeholder="7999..." onkeydown="if(event.key==='Enter') renderIdentity(window.lastOverview?.identity || {})"></label><label><span class="label">Reason</span><select id="identityAttemptFilterReason"><option value="">all</option>${reasonOptions.map(reason => `<option value="${esc(reason)}" ${reason === reasonFilter ?       const failures = allFailures.filter(a => {
        const phoneOk = !phoneFilter || String(a.phone_normalized || '').includes(phoneFilter);
        const reasonOk = !reasonFilter || (a.reason || 'unknown') === reasonFilter;
        return phoneOk && reasonOk;
      });
      if (!failures.length) return filters + '<div class="muted">No recent connect failures for this filter.</div>';
      const grouped = {};
      failures.forEach(a => {
        const reason = a.reason || 'unknown';
        grouped[reason] ||= {reason, count:0, phones:new Set(), orgs:new Set(), latest:''};
        grouped[reason].count += 1;
        if (a.phone_normalized) grouped[reason].phones.add(a.phone_normalized);
        if (a.organization_input) grouped[reason].orgs.add(a.organization_input);
        if (!grouped[reason].latest || String(a.created_at || '') > grouped[reason].latest) grouped[reason].latest = a.created_at || '';
      });
      const rows = Object.values(grouped)
        .sort((a,b) => b.count - a.count || String(b.latest).localeCompare(String(a.latest)))
        .map(row => {
          const [title, action] = connectReasonMeta(row.reason);
          return `<tr><td>${esc(title)}<br><small>${esc(row.reason)}</small></td><td>${esc(row.count)}</td><td>${esc(row.phones.size)}</td><td>${esc(row.orgs.size)}</td><td>${esc(action)}</td><td>${esc(row.latest)}</td><td><button onclick="showConnectFailures('${esc(row.reason)}')">Details</button></td></tr>`;

    function identityLifecycleMeta(user, identity) {
      const devices = (identity.devices || []).filter(d => d.user_id === user.id);
      const activeDevices = devices.filter(d => d.status === 'active');
      const deviceIps = new Set(activeDevices.map(d => d.assigned_vpn_ip).filter(Boolean));
      const deliveries = ((window.lastOverview?.profile_delivery || {}).items || []).filter(item => deviceIps.has(item.ip));
      const activeDelivery = deliveries.find(item => item.status === 'active');
      const downloadedDelivery = deliveries.find(item => item.status === 'downloaded');
      const recentFail = (identity.onboarding_attempts || []).find(a => a.phone_normalized === user.phone_normalized && a.result === 'FAIL');
      const maxDevices = Number(user.max_devices || 0);
      const activeCount = Number(user.active_devices || activeDevices.length || 0);
      if (user.status === 'blocked' || user.status === 'disabled') return {ok:false, label:user.status, action:'Unblock user before issuing new access.', deliveries};
      if (maxDevices > 0 && activeCount >= maxDevices) return {ok:false, label:'device limit', action:'Increase device limit or revoke an old device.', deliveries};
      if (!activeDevices.length && recentFail) return {ok:false, label:'needs help', action:`Latest connect failure: ${recentFail.reason || 'unknown'}.`, deliveries};
      if (!activeDevices.length) return {ok:false, label:'no device', action:'Issue the first device from this row.', deliveries};
      if (activeDelivery) return {ok:true, label:'link sent', action:'One-time link is active. Waiting for user to open/import it.', deliveries};
      if (downloadedDelivery) return {ok:true, label:'downloaded', action:'Profile was downloaded. Ask user to connect, then verify route.', deliveries};
      return {ok:true, label:'issued', action:'Device exists. Use device details to verify route if needed.', deliveries};
    }

    function identityUserEventMatch(ev, user, devices) {
      const deviceIds = new Set(devices.map(d => d.id).filter(Boolean));
      const ips = new Set(devices.map(d => d.assigned_vpn_ip).filter(Boolean));
      const text = [ev.action, ev.object_type, ev.object_id, ev.user_ip, ev.message, JSON.stringify(ev.details || {})].map(v => String(v || '')).join(' ');
      if (user.id && text.includes(user.id)) return true;
      if (user.phone_normalized && text.includes(user.phone_normalized)) return true;
      for (const id of deviceIds) if (text.includes(id)) return true;
      for (const ip of ips) if (text.includes(ip)) return true;
      return false;
    }

    function identityProvisioningQualityRows(user, devices, deliveries, lifecycle) {
      const activeDevices = (devices || []).filter(d => d.status === 'active');
      const readiness = window.lastOverview?.user_readiness || {};
      const proxyRuntime = window.lastOverview?.proxy_runtime || {};
      const activeDelivery = (deliveries || []).find(d => d.status === 'active');
      const downloadedDelivery = (deliveries || []).find(d => d.status === 'downloaded');
      const activeIps = activeDevices.map(d => d.assigned_vpn_ip).filter(Boolean);
      const readinessRows = activeIps.map(ip => readiness[ip] || {});
      const connected = readinessRows.some(r => !!r.connected);
      const routeOk = readinessRows.length ? readinessRows.some(r => !!r.route_ok && !r.leak_risk) : false;
      const ready = readinessRows.some(r => r.status === 'READY');
      const needsProxyRefresh = activeDevices.some(d => String(d.vpn_client || '').toLowerCase() === 'happ') && !!proxyRuntime.needs_refresh;
      const rows = [
        ['Identity active', user.status === 'active', user.status === 'active' ? 'user can receive devices' : `status=${user.status || 'unknown'}`],
        ['Device issued', activeDevices.length > 0, activeDevices.length ? `${activeDevices.length} active device(s)` : 'no active device yet'],
        ['Delivery link', !!activeDelivery || !!downloadedDelivery, activeDelivery ? 'one-time link is active' : (downloadedDelivery ? 'profile already downloaded' : 'no active/downloaded delivery link')],
        ['Profile downloaded', !!downloadedDelivery, downloadedDelivery ? `downloaded ${downloadedDelivery.downloaded_at || ''}` : 'waiting for user to open the link'],
        ['Client connected', connected, connected ? 'recent handshake or client agent seen' : 'waiting for first connection'],
        ['Route reality', routeOk, routeOk ? 'route check is OK for at least one active device' : 'route not verified yet'],
        ['Full readiness', ready, ready ? 'READY' : (lifecycle.action || 'waiting for checks')],
        ['Proxy runtime', !needsProxyRefresh, needsProxyRefresh ? 'happ proxy runtime refresh is required' : 'current or not required'],
      ];
      return rows.map(row => `<tr><td>${esc(row[0])}</td><td>${chip(!!row[1], row[1] ? 'OK' : 'WAIT')}</td><td>${esc(row[2])}</td></tr>`);
    }

    async function reissueIdentityDeviceProfile(deviceId) {
      const identity = window.lastOverview?.identity || {};
      const dev = (identity.devices || []).find(d => d.id === deviceId) || {};
      const ip = dev.assigned_vpn_ip || '';
      const adapter = String(dev.vpn_client || 'karing').toLowerCase();
      if (!ip) {
        alert('У этого устройства нет VPN IP.');
        return;
      }
      if (!['karing','hiddify','happ'].includes(adapter)) {
        alert(`Повторная выдача умного профиля для ${adapter || 'unknown'} пока не поддерживается.`);
        return;
      }
      const mode = prompt('Режим маршрутов для нового профиля: RU_LOCAL, ABROAD_RU_VIA_V7, AUTO_TRAVEL, STRICT_V7', 'RU_LOCAL');
      if (!mode) return;
      const cleanMode = mode.trim().toUpperCase();
      if (!confirm(`Переиздать профиль ${adapter} ${cleanMode} для ${dev.device_name || deviceId} (${ip})? Существующие VPN-ключи не меняются.`)) return;
      const el = document.getElementById('actions');
      el.textContent = `переиздаю профиль устройства для ${ip}...`;
      const profileResult = await postJson('/api/actions/smart-client-profile-generate', {ip, adapter, mode: cleanMode});
      let deliveryResult = null;
      if (profileResult.profile?.ready) {
        deliveryResult = await postJson('/api/actions/profile-delivery-create', {ip, adapter, mode: cleanMode});
      }
      el.textContent = JSON.stringify({profileResult, deliveryResult}, null, 2);
      const profile = profileResult.profile || {};
      const delivery = deliveryResult?.delivery || {};
